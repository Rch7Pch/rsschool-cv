                                function isPalindrome(line) {
                                    line += '';
                                    let str = '';
                                    for (let i = line.length - 1; i >= 0; i--) {
                                        str += line[i];
                                    }
                                    if (str === line) {
                                        return true;
                                    }
                                    return false;
                                }
